/**
 * The argument defaults.
 * @enum {Symbol}

  /** The author of the command message. */
  Author: Symbol(),

  Member: Symbol(),
  /** The channel of the command message. */
  Channel: Symbol(),
  /** The guild of the command message. */
  Guild: Symbol(),

  HighestRole: Symbol()
};

module.exports = ArgumentDefault;
